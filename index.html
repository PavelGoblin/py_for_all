Introduction Basics

Python is a high-level, interpreted, general-purpose programming language. Its simple syntax resembles English, making it ideal for beginners and powerful enough for professionals.

Why Python?

  • 🚀 Easy to learn — minimal syntax
  • 🔬 Used in Data Science, AI, Web Dev
  • 📦 Massive standard library
  • 🌍 Huge open-source community
  • 🔄 Interpreted (runs line by line)
  • 🧹 Dynamically typed
  • ⚡ Cross-platform (Win/Mac/Linux)
  • 🔗 Embeddable and extensible

Variables & Data Types Core

Variables are containers for storing data. Python is dynamically typed — you don't need to declare a type explicitly.

python

print(float(5))          # 5.0
print(str(42))           # "42"
print(bool(0))           # False (0, None, "", [], {} are falsy)
print(bool(99))          # True

F-Strings (Formatted String Literals)

python — f-strings (Python 3.6+)
name = "Alice"; age = 22; gpa = 3.876

# Basic f-string
print(f"Name: {name}, Age: {age}")

# Expressions inside {}
print(f"Next year she will be {age + 1}")

# Format specifiers
print(f"GPA: {gpa:.2f}")            # "GPA: 3.88"
print(f"{1_000_000:,}")           # "1,000,000" (thousands sep)
print(f"{name:>15}")              # "          Alice" (right-align)
print(f"{name:<15}")              # "Alice          " (left-align)
print(f"{name:^15}")              # "     Alice     " (center)
print(f"{255:08b}")             # "11111111" (binary)
print(f"{255:#x}")              # "0xff" (hex)

# f-string debug (Python 3.8+)
x = 42
print(f"{x=}")                   # "x=42" (super useful!)
Conditionals Control Flow

Python uses if, elif, and else for decision-making. Indentation is mandatory — typically 4 spaces.

falsy_values = [False, 0, 0.0, "", [], {}, (), None]
for val in falsy_values:

Loops Control Flow

Python has for loops (iterating over sequences) and while loops (repeating while a condition is true).

python — for loops
# ── Basic for loop ───────────────────────────────
for i in range(5):
    print(i, end=" ")     # 0 1 2 3 4

# range(start, stop, step)
for i in range(1, 10, 2):
    print(i, end=" ")     # 1 3 5 7 9

# ── Iterate over list ────────────────────────────
fruits = ["apple", "banana", "cherry"]
for fruit in fruits:
    print(f"  🍎 {fruit.capitalize()}")

# ── enumerate() — get index + value ──────────────
for i, fruit in enumerate(fruits, start=1):
    print(f"{i}. {fruit}")

# ── zip() — iterate multiple lists together ──────
names = ["Alice", "Bob", "Carol"]
scores = [95, 87, 92]
for name, score in zip(names, scores):
    print(f"{name}: {score}")

# ── Nested loops ─────────────────────────────────
for row in range(3):
    for col in range(3):
        print(f"({row},{col})", end=" ")
    print()   # newline after each row

# ── Loop with else ───────────────────────────────
for n in range(2, 10):
    if 7 % n == 0:
        print(f"7 is divisible by {n}"); break
else:
    print("7 is prime")    # runs only if no break occurred
python — while loops & control
# ── While loop ───────────────────────────────────
count = 0
while count < 5:
    print(count, end=" ")
    count += 1            # 0 1 2 3 4

# ── break, continue, pass ────────────────────────
for i in range(10):
    if i == 3: continue   # skip 3
    if i == 7: break      # stop at 7
    print(i, end=" ")     # 0 1 2 4 5 6

# pass = placeholder (do nothing)
for i in range(3):
    pass    # empty loop body — no error

# ── Infinite loop with break ─────────────────────
total = 0
while True:
    num = int(input("Enter number (0 to stop): "))
    if num == 0: break
    total += num
print(f"Total: {total}")

Functions Functions

Functions are reusable blocks of code defined with def. Python supports default args, *args, **kwargs, and type hints.

python — comprehensive function guide
# ── Basic Function ───────────────────────────────
def greet(name):
    """Returns a greeting string."""  # docstring
    return f"Hello, {name}!"

print(greet("Alice"))             # "Hello, Alice!"
print(greet.__doc__)              # access docstring

# ── Default Arguments ────────────────────────────
def power(base, exp=2):
    return base ** exp

print(power(3))        # 9   (exp defaults to 2)
print(power(2, 10))    # 1024

# ── Keyword Arguments ────────────────────────────
def student_info(name, age, city="Unknown"):
    return f"{name}, {age}, from {city}"

print(student_info(age=21, name="Bob", city="Dhaka"))

# ── *args (variable positional args) ─────────────
def add_all(*nums):
    print(type(nums))           # <class 'tuple'>
    return sum(nums)

print(add_all(1, 2, 3, 4, 5))   # 15

# ── **kwargs (variable keyword args) ─────────────
def profile(**info):
    print(type(info))           # <class 'dict'>
    for k, v in info.items():
        print(f"  {k}: {v}")

profile(name="Alice", age=22, gpa=3.9)

# ── Combined: everything together ────────────────
def mega_func(a, b, *args, key="default", **kwargs):
    print(f"a={a}, b={b}, extra={args}, key={key}, rest={kwargs}")

mega_func(1, 2, 3, 4, key="hello", x=10, y=20)

# ── Type Hints (Python 3.5+) ─────────────────────
def multiply(a: int, b: int) -> int:
    return a * b

# ── Multiple Return Values ───────────────────────
def min_max(numbers: list) -> tuple:
    return min(numbers), max(numbers)

lo, hi = min_max([3, 1, 8, 4])
print(f"Min: {lo}, Max: {hi}")      # Min: 1, Max: 8

# ── Scope: LEGB Rule ─────────────────────────────
x = "global"
def outer():
    x = "enclosing"
    def inner():
        x = "local"
        print(x)    # "local"
    inner()
    print(x)        # "enclosing"
outer()
print(x)            # "global"
Lambda & Higher-Order Functions Functional

Lambda functions are anonymous one-liners. Higher-order functions like map(), filter(), and sorted() take functions as arguments.

python
# ── Lambda Syntax: lambda args: expression ────────
square = lambda x: x ** 2
add = lambda x, y: x + y
clamp = lambda x, lo, hi: max(lo, min(x, hi))

print(square(5))             # 25
print(add(3, 4))             # 7
print(clamp(15, 0, 10))     # 10

# ── map(func, iterable) ──────────────────────────
nums = [1, 2, 3, 4, 5]
squares = list(map(lambda x: x**2, nums))
print(squares)     # [1, 4, 9, 16, 25]

# ── filter(func, iterable) ───────────────────────
evens = list(filter(lambda x: x % 2 == 0, nums))
print(evens)       # [2, 4]

# ── sorted with key ──────────────────────────────
students = [("Alice", 22), ("Bob", 19), ("Carol", 25)]
by_age = sorted(students, key=lambda s: s[1])
print(by_age)      # [('Bob',19), ('Alice',22), ('Carol',25)]

# ── reduce (functools) ───────────────────────────
from functools import reduce
product = reduce(lambda a, b: a * b, [1,2,3,4,5])
print(product)     # 120 (5! = 120)

# ── Functions as arguments ───────────────────────
def apply_twice(func, x):
    return func(func(x))

print(apply_twice(lambda x: x + 3, 10))  # 16

# ── Closure ──────────────────────────────────────
def make_multiplier(n):
    return lambda x: x * n     # captures n

double = make_multiplier(2)
triple = make_multiplier(3)
print(double(5), triple(5))   # 10 15
Classes & OOP OOP

Object-Oriented Programming organizes code into classes (blueprints) and instances (objects). Python supports full OOP with encapsulation, inheritance, and polymorphism.

python — complete class example
class BankAccount:
    """A simple bank account class."""

    bank_name = "PyBank"     # class variable (shared by all)
    _count = 0               # private-ish by convention

    def __init__(self, owner: str, balance: float = 0.0):
        self.owner = owner       # instance variable
        self.__balance = balance # name-mangled (truly private)
        BankAccount._count += 1

    # ── Instance Methods ─────────────────────────────
    def deposit(self, amount: float):
        if amount <= 0:
            raise ValueError("Amount must be positive")
        self.__balance += amount
        return self   # enable method chaining!

    def withdraw(self, amount: float):
        if amount > self.__balance:
            raise ValueError("Insufficient funds")
        self.__balance -= amount
        return self

    # ── Property (getter) ────────────────────────────
    @property
    def balance(self) -> float:
        return self.__balance

    # ── Class Method ─────────────────────────────────
    @classmethod
    def get_count(cls) -> int:
        return cls._count

    # ── Static Method ────────────────────────────────
    @staticmethod
    def is_valid_amount(amount: float) -> bool:
        return isinstance(amount, (int, float)) and amount > 0

    # ── Dunder Methods ───────────────────────────────
    def __str__(self) -> str:     # print(obj)
        return f"{self.owner}'s account: ৳{self.__balance:.2f}"

    def __repr__(self) -> str:    # developer repr
        return f"BankAccount('{self.owner}', {self.__balance})"

    def __add__(self, other):     # acc1 + acc2
        return BankAccount(f"{self.owner}&{other.owner}",
                           self.__balance + other.__balance)

# ── Using the class ──────────────────────────────
acc = BankAccount("Alice", 1000)
acc.deposit(500).deposit(200).withdraw(100)  # chaining
print(acc)                    # Alice's account: ৳1600.00
print(acc.balance)            # 1600.0 (via property)
print(BankAccount.get_count()) # 1 (class method)
print(BankAccount.is_valid_amount(-5)) # False (static)
Inheritance OOP

Inheritance lets a class acquire attributes and methods of another class, promoting code reuse and the "is-a" relationship.

python
# ── Base Class ───────────────────────────────────
class Animal:
    def __init__(self, name: str, sound: str):
        self.name = name
        self.sound = sound

    def speak(self):
        return f"{self.name} says '{self.sound}'"

    def __str__(self):
        return f"{self.__class__.__name__}({self.name})"

# ── Single Inheritance ───────────────────────────
class Dog(Animal):
    def __init__(self, name: str, breed: str):
        super().__init__(name, "Woof")  # call parent
        self.breed = breed

    def fetch(self, item: str):          # new method
        return f"{self.name} fetches the {item}!"

    def speak(self):                     # override
        return super().speak() + " 🐶"

class Cat(Animal):
    def __init__(self, name: str, indoor: bool = True):
        super().__init__(name, "Meow")
        self.indoor = indoor

# ── Polymorphism ─────────────────────────────────
animals = [Dog("Rex", "German Shepherd"), Cat("Whiskers")]
for a in animals:
    print(a.speak())         # each class overrides speak()

# ── Multiple Inheritance ─────────────────────────
class Flyable:
    def fly(self): return "I can fly!"

class Swimmable:
    def swim(self): return "I can swim!"

class Duck(Animal, Flyable, Swimmable):
    def __init__(self): super().__init__("Duck", "Quack")

d = Duck()
print(d.speak(), d.fly(), d.swim())

# ── Abstract Classes ─────────────────────────────
from abc import ABC, abstractmethod

class Shape(ABC):
    @abstractmethod
    def area(self) -> float: ...    # must be implemented

    @abstractmethod
    def perimeter(self) -> float: ...

class Circle(Shape):
    def __init__(self, r): self.r = r
    def area(self): return 3.14159 * self.r**2
    def perimeter(self): return 2 * 3.14159 * self.r
Comprehensions Advanced

Comprehensions are a concise, Pythonic way to create lists, dicts, and sets from iterables — often faster than equivalent loops.

python
# ── List Comprehension: [expr for item in iter if cond]
squares = [x**2 for x in range(10)]
print(squares)      # [0,1,4,9,16,25,36,49,64,81]

evens = [x for x in range(20) if x % 2 == 0]
words = ["  hello  ", "WORLD", " Python "]
clean = [w.strip().lower() for w in words]

# Nested list comprehension (flatten matrix)
matrix = [[1,2,3],[4,5,6],[7,8,9]]
flat = [n for row in matrix for n in row]
print(flat)         # [1,2,3,4,5,6,7,8,9]

# ── Dict Comprehension: {k:v for ...} ────────────
nums = [1, 2, 3, 4, 5]
sq_dict = {n: n**2 for n in nums}
print(sq_dict)       # {1:1, 2:4, 3:9, 4:16, 5:25}

# Swap keys and values
d = {"a":1, "b":2, "c":3}
inverted = {v: k for k, v in d.items()}

# ── Set Comprehension: {expr for ...} ────────────
unique_lens = {len(w) for w in ["hi","hello","hey","hi"]}
print(unique_lens)   # {2, 3, 5}

# ── Generator Expression: (expr for ...) ─────────
# Lazy — does NOT store all values in memory
gen = (x**2 for x in range(1_000_000))  # no memory spike!
print(next(gen))     # 0 (get first value)
print(sum(x**2 for x in range(10)))  # 285 (no brackets needed!)
Exception Handling Control Flow

Exceptions handle errors gracefully at runtime. Python uses try/except/else/finally blocks.

python
# ── Basic try/except ─────────────────────────────
try:
    result = 10 / 0
except ZeroDivisionError:
    print("Cannot divide by zero!")

# ── Multiple exceptions ──────────────────────────
def safe_convert(val):
    try:
        num = int(val)
        result = 100 / num
        return result
    except ValueError:
        return "Not a number"
    except ZeroDivisionError:
        return "Cannot be zero"
    except (TypeError, OverflowError) as e:
        return f"Error: {e}"

# ── else & finally ───────────────────────────────
try:
    f = open("data.txt")
    content = f.read()
except FileNotFoundError as e:
    print(f"File missing: {e}")
else:
    print("Read successfully!")   # runs only if no error
    print(content[:100])
finally:
    print("This always runs")    # cleanup code

# ── Custom Exceptions ────────────────────────────
class InsufficientFundsError(Exception):
    def __init__(self, amount, balance):
        self.amount = amount
        self.balance = balance
        super().__init__(f"Need ৳{amount} but have ৳{balance}")

def withdraw(balance, amount):
    if amount > balance:
        raise InsufficientFundsError(amount, balance)
    return balance - amount

try:
    withdraw(500, 1000)
except InsufficientFundsError as e:
    print(e)                   # "Need ৳1000 but have ৳500"

Decorators Advanced

Decorators wrap a function to add behavior before/after it runs — using the @decorator syntax.

python
import functools, time

# ── Basic Decorator ──────────────────────────────
def logger(func):
    @functools.wraps(func)         # preserve func metadata
    def wrapper(*args, **kwargs):
        print(f"▶ Calling {func.__name__}()")
        result = func(*args, **kwargs)
        print(f"✔ {func.__name__} returned: {result}")
        return result
    return wrapper

@logger
def add(a, b): return a + b

add(3, 4)   # ▶ Calling add() ... ✔ add returned: 7

# ── Timer Decorator ──────────────────────────────
def timer(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        end = time.perf_counter()
        print(f"⏱ {func.__name__} took {end-start:.4f}s")
        return result
    return wrapper

@timer
def slow_func():
    time.sleep(0.1)
    return "done"

# ── Decorator with Arguments ─────────────────────
def repeat(n):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for _ in range(n):
                func(*args, **kwargs)
        return wrapper
    return decorator

@repeat(3)
def say_hi(): print("Hi!")

say_hi()  # Hi! Hi! Hi!

# ── Stacking Decorators ──────────────────────────
@timer
@logger
def process(data): return data * 2
# applied bottom-up: logger wraps process, then timer wraps that
File I/O Advanced

Reading and writing files is fundamental. Python's built-in open() function and the with statement make it clean and safe.

python
# ── Writing Files ────────────────────────────────
with open("notes.txt", "w", encoding="utf-8") as f:
    f.write("Hello, File!\n")
    f.write("Python is great\n")
    f.writelines(["Line 3\n", "Line 4\n"])

# ── Reading Files ────────────────────────────────
with open("notes.txt", "r", encoding="utf-8") as f:
    content = f.read()             # entire file as string

with open("notes.txt") as f:
    lines = f.readlines()         # list of lines (with \n)

with open("notes.txt") as f:
    for line in f:                # memory-efficient iteration
        print(line.strip())

# ── Modes ────────────────────────────────────────
# "r"  = read (default)
# "w"  = write (overwrites!)
# "a"  = append (adds to end)
# "x"  = create (fails if exists)
# "rb" = read binary (images, etc.)
# "r+" = read and write

# ── JSON Files ───────────────────────────────────
import json

data = {"name": "Alice", "scores": [90, 85, 92]}

with open("data.json", "w") as f:
    json.dump(data, f, indent=4)  # write JSON

with open("data.json") as f:
    loaded = json.load(f)        # read JSON

# JSON string conversion
s = json.dumps(data, indent=2)  # dict → JSON string
d = json.loads(s)               # JSON string → dict
